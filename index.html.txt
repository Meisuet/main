// 更新時鐘 (每秒更新)
function updateClock() {
const now = new Date();
const year = now.getFullYear();
const month = String(now.getMonth() + 1).padStart(2, '0');
const date = String(now.getDate()).padStart(2, '0');
const weekdays = ['SUN', 'MON', 'TUE', 'WED', 'THU', 'FRI', 'SAT'];
const day = weekdays[now.getDay()];

let hours = now.getHours();
const ampm = hours >= 12 ? 'PM' : 'AM';
hours = hours % 12;
hours = hours ? hours : 12; // 0 點轉為 12 點
const minutes = String(now.getMinutes()).padStart(2, '0');
const seconds = String(now.getSeconds()).padStart(2, '0');

document.getElementById('current-date').innerText = `${year}/${month}/${date} ${day}`;
document.getElementById('current-time').innerText = `${String(hours).padStart(2, '0')}:${minutes}:${seconds} ${ampm}`;
}
setInterval(updateClock, 1000);
updateClock();

// 獲取交通工具的 ETA (天瑞站)
// 註：天瑞站的輕鐵月台代號或巴士站代號需與港鐵 API 對齊。
// 天瑞輕鐵站編號一般為 460 (705在1月台，761P在2月台，此處以常用API邏輯示範)
async function fetchTransportETA() {
try {
// 1. 輕鐵 705 及 761P (天瑞站 ID: 460)
const lrResponse = await fetch('https://rt.data.gov.hk/v1/transport/mtr/lrt/getSchedule?stationId=460');
const lrData = await lrResponse.json();

if(lrData.status === 1 && lrData.platform_list) {
processLRT(lrData.platform_list);
}
} catch (e) {
console.error("無法獲取輕鐵數據，這通常是由於瀏覽器 CORS 限制引起。", e);
}

try {
// 2. 港鐵巴士 K75P (天瑞總站/分站編號，此處以 DATA.GOV.HK 港鐵巴士 API 格式為準)
// 備註：部分 MTR 巴士需要傳入特定的 route 與 station 參數
const busResponse = await fetch('https://rt.data.gov.hk/v1/transport/mtr/bus/getSchedule?route=K75P');
const busData = await busResponse.json();
if(busData.status === 1) {
processBus(busData.bus_list, 'K75P');
}
} catch (e) {
console.error("無法獲取巴士數據", e);
}
}

// 處理輕鐵數據
function processLRT(platforms) {
let eta705 = [];
let eta761p = [];

platforms.forEach(platform => {
if(platform.route_list) {
platform.route_list.forEach(route => {
// 過濾兩班車
if(route.route_no === '705') {
eta705.push(route.time_en || route.time_ch);
}
if(route.route_no === '761P') {
eta761p.push(route.time_en || route.time_ch);
}
});
}
});

// 更新網頁顯示 (取前兩班)
renderETADisplay('eta-705', eta705.slice(0, 2));
renderETADisplay('eta-761p', eta761p.slice(0, 2));
}

// 處理港鐵巴士數據
function processBus(busList, routeNo) {
// 篩選出屬於天瑞站(依據你的具體需求篩選 station_id)往洪水橋方向的車次
// 這裡簡化為抓取該路線前兩班倒數時間
let etaBus = [];
if(busList && busList[routeNo]) {
busList[routeNo].forEach(bus => {
// 計算剩餘分鐘數
if(bus.departure_time) {
const diffMs = new Date(bus.departure_time) - new Date();
const diffMins = Math.max(0, Math.floor(diffMs / 60000));
etaBus.push(`${diffMins} 分鐘`);
}
});
}
renderETADisplay('eta-k75p', etaBus.slice(0, 2));
}

function renderETADisplay(elementId, etaArray) {
const container = document.getElementById(elementId);
if(etaArray.length === 0) {
container.innerHTML = `<span class="eta-time">暫無班次</span>`;
return;
}
container.innerHTML = etaArray.map(time => `<span class="eta-time">${time}</span>`).join('');
}

// 獲取香港天文台天氣數據
async function fetchWeatherData() {
try {
// 現時天氣資料
const weatherResp = await fetch('https://data.weather.gov.hk/weatherAPI/opendata/weather.php?dataType=rhrread&lang=tc');
const weatherData = await weatherResp.json();

// 1. 溫度 (取濕地公園或屯門/元朗最接近天水圍的觀測站，這裡用濕地公園作示範，若無則用總部)
const wetLandParkTemp = weatherData.temperature.data.find(item => item.place === '濕地公園');
const tsimShaTsuiTemp = weatherData.temperature.data.find(item => item.place === '香港天文台');
const currentTemp = wetLandParkTemp ? wetLandParkTemp.value : tsimShaTsuiTemp.value;
document.getElementById('weather-temp').innerText = `${currentTemp} °C`;

// 2. 濕度
const currentHumidity = weatherData.humidity.data[0].value;
document.getElementById('weather-humidity').innerText = `濕度: ${currentHumidity}%`;

// 3. 紫外光指數
if(weatherData.uvindex && weatherData.uvindex.data && weatherData.uvindex.data.length > 0) {
document.getElementById('weather-uv').innerText = `UV 指數: ${weatherData.uvindex.data[0].value}`;
} else {
document.getElementById('weather-uv').innerText = `UV 指數: 0`;
}

// 4. 天氣警告提示
const warningResp = await fetch('https://data.weather.gov.hk/weatherAPI/opendata/weather.php?dataType=warnsum&lang=tc');
const warningData = await warningResp.json();
const alertEl = document.getElementById('weather-alert');

if (Object.keys(warningData).length > 0) {
// 串接所有生效中的警告名稱
const warnings = Object.values(warningData).map(w => w.name).join(' | ');
alertEl.innerText = `警告: ${warnings}`;
alertEl.style.display = 'block';
} else {
alertEl.style.display = 'none';
}

} catch (e) {
console.error("無法獲取天氣數據", e);
}
}

// 定時更新設定 (每 60 秒更新一次數據)
function initDataFetch() {
fetchTransportETA();
fetchWeatherData();
}

setInterval(initDataFetch, 60000);
initDataFetch(); // 初始化載入
